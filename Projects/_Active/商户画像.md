```dataviewjs
// 取当前项目的链接（如 [[示例项目]]）
const cur = dv.current().file.link;

// 抓取所有包含当前项目链接的任务
let tasks = [];
for (const p of dv.pages().where(p => p.file.tasks)) {
  for (const t of p.file.tasks) {
    const txt = (t.text || "") + " " + (t.link?.path || "");
    if (txt.includes(cur.path) || txt.includes(cur.display)) tasks.push(t);
  }
}

// 按 due 升序
const byDue = (a,b) => {
  const ad = a.due ? moment(a.due) : null;
  const bd = b.due ? moment(b.due) : null;
  if (!ad && !bd) return 0;
  if (!ad) return 1;
  if (!bd) return -1;
  return ad.diff(bd);
};
tasks.sort(byDue);

// 输出任务列表（带复选框、可在此处直接勾选）
dv.taskList(tasks, true);
```
